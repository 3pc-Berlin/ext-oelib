  32.  * @package        TYPO3
  33.  * @subpackage    tx_oelib

  37. class tx_oelib_timer {
  38.     /**
  39.      * @var boolean whether the timer currently is running
  40.      */
  41.     private $isRunning = false;
  42.  
  43.     /**
  44.      * @var string the name of the current time bucket

  46.     private $currentBucketName = '';

  48.     /**
  49.      * @var array a stack of previously used buckets, starting with the first

  52.     private $previousBucketNames = array();

  54.     /**

  62.     private $allTime = 0.00;

  67.     private $lastTime = 0.00;

  78.     }

  84.         $this->destroyAllBuckets();
  85.     }

  87.     /**
  88.      * Retrieves the singleton timer instance. This function usually should be

  93.     public static function getInstance() {
  94.         if (!self::$instance) {
  95.             self::$instance = new tx_oelib_timer();

  99.     }

  101.     /**
  102.      * Starts the timer and sets the bucket which the time will be added to.
  103.      *
  104.      * If the timer is already running, the previous bucket will be closed
  105.      * first.
  106.      *
  107.      * @param string the name of the bucket to open, must not be empty
  108.      */
  109.     public function openBucket($bucketName = 'default') {
  110.         if ($bucketName != $this->currentBucketName) {
  111.             $this->closeCurrentBucket();
  112.             $this->previousBucketNames[] = $this->currentBucketName;
  113.             $this->currentBucketName = $bucketName;
  114.         }
  115.  
  116.         $this->isRunning = true;
  117.     }
  118.  
  119.     /**
  120.      * Starts the timer and sets the bucket which the time will be added to.
  121.      *
  122.      * If the timer is already running, the previous bucket will be closed
  123.      * first.
  124.      *
  125.      * This is a static shortcut for openBucket.
  126.      *
  127.      * @see openBucket
  128.      *
  129.      * @param string the name of the bucket to open, must not be empty
  130.      */
  131.     public static function oB($bucketName = 'default') {
  132.         tx_oelib_timer::getInstance()->openBucket($bucketName);
  133.     }
  134.  
  135.     /**
  136.      * Stops the timer and adds the passed time to the current bucket.
  137.      */
  138.     public function stopTimer() {
  139.         $this->closeCurrentBucket();
  140.         $this->isRunning = false;
  141.     }
  142.  
  143.     /**
  144.      * Gets the statistics as an arry sorted descending by time.
  145.      * Each array element will be an array in itself, containing the following
  146.      * keys:
  147.      * - bucketName (string, not htmlspecialchared yet)
  148.      * - absoluteTime (float, in seconds)
  149.      * - relativeTime (float, in percent)
  150.      *
  151.      * @return array two-dimensional array with the times of all buckets,
  152.      *                will be empty if there are no buckets
  153.      */
  154.     public function getStatisticsAsRawData() {
  155.         $this->stopTimer();
  156.         $this->clearAllPreviousBuckets();
  157.  
  158.         // Put the biggest performance culprits on top of the list.
  159.         arsort($this->buckets, SORT_NUMERIC);
  160.  
  161.         $result = array();
  162.  
  163.         foreach ($this->buckets as $bucketName => $bucketTime) {
  164.             $result[] = array(
  165.                 'bucketName' => $bucketName,
  166.                 'absoluteTime' => $bucketTime,
  167.                 'relativeTime' => $bucketTime * 100 / $this->allTime
  168.             );
  169.         }
  170.  
  171.         return $result;
  172.     }
  173.  
  174.     /**
  175.      * Gets the statistics as a HTML table with one row per bucket, containing
  176.      * the bucket name, the absolute amount of time in seconds in the bucket
  177.      * and the percentage of the total time (from all buckets summed up).
  178.      *
  179.      * The table will be sorted with the biggest buckets first.
  180.      *
  181.      * @return string a HTML table with the statistics
  182.      */
  183.     public function getStatistics() {
  184.         $rawStatistics = $this->getStatisticsAsRawData();
  185.  
  186.         $result .= '<table summary="statistics">'.LF;
  187.         $result .= '  <thead>'.LF;
  188.         $result .= '    <tr>'.LF;
  189.         $result .= '      <th scope="col">Total time</th>'
  190.             .'<th scope="col">'.$this->allTime
  191.             .'&nbsp;s</th><th>100&nbsp;%</th>'.LF;
  192.         $result .= '    </tr>'.LF;
  193.         $result .= '  </thead>'.LF;
  194.         $result .= '  <tbody>'.LF;
  195.  
  196.         foreach ($rawStatistics as $bucketData) {
  197.             $result .= '    <tr>'.LF;
  198.             $result .= '      <td>'.htmlspecialchars($bucketData['bucketName'])
  199.                 .'</td>';
  200.             $result .= '      <td>'.$bucketData['absoluteTime'].'&nbsp;s'.'</td>';
  201.             $result .= '      <td>'.$bucketData['relativeTime'].'&nbsp;%'
  202.                 .'</td>'.LF;
  203.             $result .= '    </tr>'.LF;
  204.         }
  205.  
  206.         $result .= '  </tbody>'.LF;
  207.         $result .= '</table>'.LF;
  208.  
  209.         return $result;
  210.     }

  212.     /**
  213.      * Stops all timers and deletes all buckets.
  214.      *
  215.      * After this, a completely new sets of buckets can be created.
  216.      */
  217.     public function destroyAllBuckets() {
  218.         $this->stopTimer();
  219.  
  220.         foreach ($this->buckets as $bucketKey => $bucket) {
  221.             unset($this->buckets[$bucketKey]);
  222.         }
  223.     }
  224.  
  225.     /**
  226.      * Closes the current bucket (if the timer is running), adds the passed
  227.      * time to it and set $this->lastTime to the current time.
  228.      *
  229.      * Note: This function does not stop the timer.
  230.      */
  231.     private function closeCurrentBucket() {
  232.         $currentTime = microtime(true);
  233.  
  234.         if ($this->isRunning) {
  235.             $usedTime = $currentTime - $this->lastTime;
  236.             $this->buckets[$this->currentBucketName] += $usedTime;
  237.             $this->allTime += $usedTime;
  238.         }
  239.  
  240.         $this->lastTime = $currentTime;
  241.     }
  242.  
  243.     /**
  244.      * Closes the current bucker and returns to the previous bucket (from the
  245.      * stack of previously used buckets).
  246.      *
  247.      * If there is not previous bucket, the timer will be stopped.
  248.      */
  249.     public function returnToPreviousBucket() {
  250.         $this->closeCurrentBucket();
  251.         $previousBucketName = array_pop($this->previousBucketNames);
  252.  
  253.         if ($previousBucketName !== null) {
  254.             $this->currentBucketName = $previousBucketName;
  255.             $this->isRunning = true;
  256.         } else {
  257.             $this->stopTimer();
  258.         }
  259.     }
  260.  
  261.     /**
  262.      * Closes the current bucker and returns to the previous bucket (from the
  263.      * stack of previously used buckets).
  264.      *
  265.      * If there is not previous bucket, the timer will be stopped.
  266.      *
  267.      * This is a static shortcut for returnToPreviousBucket.
  268.      *
  269.      * @see returnToPreviousBucket
  270.      */
  271.     public static function rB() {
  272.         tx_oelib_timer::getInstance()->returnToPreviousBucket();
  273.     }
  274.  
  275.     /**
  276.      * Empties the stack of previous buckets.
  277.      */
  278.     private function clearAllPreviousBuckets() {
  279.         $this->previousBucketNames = array();
  280.     }
