  27. /**
  28.  * Class 'tx_oelib_emailCollector' for the 'oelib' extension.
  29.  *
  30.  * This class stores all parameters which were meant to be sent as an e-mail and
  31.  * provides various functions to get them for testing purposes.
  32.  * Regarding the Strategy pattern, sendEmail() represents one concrete behavior.
  33.  *
  34.  * @package        TYPO3
  35.  * @subpackage    tx_oelib
  36.  *
  37.  * @author        Saskia Metzler <saskia@merlin.owl.de>
  38.  */
  39. class tx_oelib_emailCollector extends tx_oelib_abstractMailer {
  40.     /**
  41.      * Two-dimensional array of e-mail data.
  42.      * Each e-mail is stored in one element. So the number of elements in the
  43.      * first dimension depends on how many e-mails are currently stored. One
  44.      * stored e-mail is always an associative array with four elements named
  45.      * 'recipient', 'subject', 'message' and 'headers'.
  46.      */
  47.     private $emailData = array();
  48.  
  49.     /** whether sendEmail() should return true or false */
  50.     private $fakeSuccess = true;
  51.  
  52.     /**
  53.      * Stores the contents which were meant to be sent as an e-mail.
  54.      *
  55.      * @param    string        the recipient's e-mail address, will not be
  56.      *                          validated, must not be empty
  57.      * @param    string        e-mail subject, must not be empty
  58.      * @param    string        message to send, must not be empty
  59.      * @param    string        headers, separated by linefeed, may be empty
  60.      * @param    string        encoding type: "base64", "quoted-printable" or "8bit"
  61.      * @param    string        charset to use for encoding headers (only if
  62.      *                          $encodingType is set to a valid value which produces
  63.      *                          such a header)
  64.      * @param    boolean        if set, the header content will not be encoded
  65.      *
  66.      * @param    boolean        depending on whether success should be faked or not
  67.      */
  68.     public function sendEmail(
  69.         $emailAddress,
  70.         $subject,
  71.         $message,
  72.         $headers = '',
  73.         $encodingType = '',
  74.         $charset = '',
  75.         $doNotEncodeHeader = false
  76.     ) {
  77.         $this->emailData[] = array(
  78.             'recipient' => $emailAddress,
  79.             'subject' => $subject,
  80.             'message' => $this->formatEmailBody($message),
  81.             'headers' => $headers
  82.         );
  83.  
  84.         return $this->fakeSuccess;
  85.     }
  86.  
  87.     /**
  88.      * Sets the return value for sendEmail().
  89.      *
  90.      * @param    boolean        true if sendEmail() should return true, false
  91.      *                          otherwise
  92.      */
  93.     public function setFakedReturnValue($isSuccessful) {
  94.         $this->fakeSuccess = $isSuccessful;
  95.     }
  96.  
  97.     /**
  98.      * Returns the last e-mail or an empty array if there is none.
  99.      *
  100.      * @return    array        e-mail address, subject, message and headers of the
  101.      *                          last e-mail in an array, will be empty if there is
  102.      *                          no e-mail
  103.      */
  104.     public function getLastEmail() {
  105.         if (empty($this->emailData)) {
  106.             return array();
  107.         }
  108.  
  109.         return end($this->emailData);
  110.     }
  111.  
  112.     /**
  113.      * Returns all e-mails sent with this instance or an empty array if there is
  114.      * none.
  115.      *
  116.      * @return    array        two-dimensional array with one element for each
  117.      *                          e-mail, each inner array has four elements
  118.      *                          'recipient', 'subject', 'message' and 'headers',
  119.      *                          will be empty if there are no e-mails
  120.      *
  121.      * @see        emailData
  122.      */
  123.     public function getAllEmail() {
  124.         return $this->emailData;
  125.     }
  126.  
  127.     /**
  128.      * Returns the last e-mail's recipient.
  129.      *
  130.      * @return    string        recipient of the last sent e-mail or an empty string
  131.      *                          if there is none
  132.      */
  133.     public function getLastRecipient() {
  134.         return $this->getElementFromLastEmail('recipient');
  135.     }
  136.  
  137.     /**
  138.      * Returns the last e-mail's subject.
  139.      *
  140.      * @return    string        subject of the last sent e-mail or an empty string
  141.      *                          if there is none
  142.      */
  143.     public function getLastSubject() {
  144.         return $this->getElementFromLastEmail('subject');
  145.     }
  146.  
  147.     /**
  148.      * Returns the last e-mail's body.
  149.      *
  150.      * @return    string        body of the last sent e-mail or an empty string if
  151.      *                          there is none
  152.      */
  153.     public function getLastBody() {
  154.         return $this->getElementFromLastEmail('message');
  155.     }
  156.  
  157.     /**
  158.      * Returns the last e-mail's additional headers.
  159.      *
  160.      * @return    string        headers of the last sent e-mail or an empty string
  161.      *                          if there are none
  162.      */
  163.     public function getLastHeaders() {
  164.         return $this->getElementFromLastEmail('headers');
  165.     }
  166.  
  167.     /**
  168.      * Returns an element from the array with the last e-mail.
  169.      *
  170.      * @param    string        key of the element to return, must be 'recipient',
  171.      *                          'subject', 'message' or 'headers'
  172.      *
  173.      * @return    string        value of the element, will be an empty string if
  174.      *                          there was none
  175.      */
  176.     private function getElementFromLastEmail($key) {
  177.         if (!in_array($key, array('recipient', 'subject', 'message', 'headers'))) {
  178.             throw new Exception(
  179.                 'The key "'.$key.'" is invalid. '
  180.                     .'It must be "recipient", "subject", "message" or "headers".'
  181.             );
  182.         }
  183.         if (empty($this->emailData)) {
  184.             return '';
  185.         }
  186.  
  187.         $lastEmail = $this->getLastEmail();
  188.  
  189.         return $lastEmail[$key];
  190.     }
  191. }
