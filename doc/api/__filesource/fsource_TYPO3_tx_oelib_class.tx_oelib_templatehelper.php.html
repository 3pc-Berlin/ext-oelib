  463.     /**
  464.      * Retrieves the plugin template file set in $this->conf['templateFile'] (or
  465.      * also via flexforms if TYPO3 mode is FE) and writes it to
  466.      * $this->templateCode. The subparts will be written to $this->templateCache.
  467.      *
  468.      * @param    boolean        whether the settings in the Flexform should be
  469.      *                          ignored
  470.      *
  471.      * @access    protected
  472.      */
  473.     function getTemplateCode($ignoreFlexform = false) {
  474.         // Trying to fetch the template code via $this->cObj in BE mode leads to
  475.         // a non-catchable error in the tslib_content class because the cObj
  476.         // configuration array is not initialized properly.
  477.         // As flexforms can be used in FE mode only, $ignoreFlexform is set true
  478.         // if we are in the BE mode. By this, $this->cObj->fileResource can be
  479.         // sheltered from being called.
  480.         if (TYPO3_MODE == 'BE') {
  481.             $ignoreFlexform = true;
  482.         }
  483.  
  484.         $templateFileName = $this->getConfValueString(
  485.             'templateFile',
  486.             's_template_special',
  487.             true,
  488.             $ignoreFlexform
  489.         );
  490.  
  491.         if (!$ignoreFlexform) {
  492.             $templateRawCode = $this->cObj->fileResource($templateFileName);
  493.         } else {
  494.             // If there is no need to care about flexforms, the template file is
  495.             // fetched directly from the local configuration array.
  496.             $templateRawCode = file_get_contents(
  497.                 t3lib_div::getFileAbsFileName($templateFileName)
  498.             );
  499.         }
  500.  
  501.         $this->processTemplate($templateRawCode);
  502.     }
  503.  
  504.     /**
  505.      * Stores the given HTML template and retrieves all subparts, writing them
  506.      * to $this->templateCache.
  507.      *
  508.      * The subpart names are automatically retrieved from $templateRawCode and
  509.      * are used as array keys. For this, the ### are removed, but the names stay
  510.      * uppercase.
  511.      *
  512.      * Example: The subpart ###MY_SUBPART### will be stored with the array key
  513.      * 'MY_SUBPART'.
  514.      *
  515.      * @param    string        the content of the HTML template
  516.      *
  517.      * @access    protected
  518.      */
  519.     function processTemplate($templateRawCode) {
  520.         $this->templateCode = $templateRawCode;
  521.         $this->markerNames = $this->findMarkers();
  522.  
  523.         $subpartNames = $this->findSubparts();
  524.  
  525.         foreach ($subpartNames as $subpartName) {
  526.             $matches = array();
  527.             preg_match(
  528.                 '/<!-- *###'.$subpartName.'### *-->(.*)'
  529.                     .'<!-- *###'.$subpartName.'### *-->/msSU',
  530.                 $templateRawCode,
  531.                 $matches
  532.             );
  533.             if (isset($matches[1])) {
  534.                 $this->templateCache[$subpartName] = $matches[1];
  535.             }
  536.         }
  537.     }
  538.  
  539.     /**
  540.      * Finds all subparts within the current HTML template.
  541.      * The subparts must be within HTML comments.
  542.      *
  543.      * @return    array        a list of the subpart names (uppercase, without ###,
  544.      *                         for example 'MY_SUBPART')
  545.      *
  546.      * @access    protected
  547.      */
  548.     function findSubparts() {
  549.         $matches = array();
  550.         preg_match_all(
  551.             '/<!-- *(###)([A-Z]([A-Z0-9_]*[A-Z0-9])?)(###)/',
  552.             $this->templateCode,
  553.             $matches
  554.         );
  555.  
  556.         return array_unique($matches[2]);
  557.     }
  558.  
  559.     /**
  560.      * Finds all markers within the current HTML template.
  561.      * Note: This also finds subpart names.
  562.      *
  563.      * The result is one long string that is easy to process using regular
  564.      * expressions.
  565.      *
  566.      * Example: If the markers ###FOO### and ###BAR### are found, the string
  567.      * "#FOO#BAR#" would be returned.
  568.      *
  569.      * @return    string        a list of markes as one long string, separated,
  570.      *                         prefixed and postfixed by '#'
  571.      */
  572.     private function findMarkers() {
  573.         $matches = array();
  574.         preg_match_all(
  575.             '/(###)(([A-Z0-9_]*[A-Z0-9])?)(###)/', $this->templateCode, $matches
  576.         );
  577.  
  578.         $markerNames = array_unique($matches[2]);
  579.  
  580.         return '#'.implode('#', $markerNames).'#';
  581.     }
  582.  
  583.     /**
  584.      * Gets a list of markers with a given prefix.
  585.      * Example: If the prefix is "WRAPPER" (or "wrapper", case is not relevant),
  586.      * the following array might be returned: ("WRAPPER_FOO", "WRAPPER_BAR")
  587.      *
  588.      * If there are no matches, an empty array is returned.
  589.      *
  590.      * The function <code>findMarkers</code> must be called before this function
  591.      * may be called.
  592.      *
  593.      * @param    string    case-insensitive prefix for the marker names to look for
  594.      *
  595.      * @return    array    array of matching marker names, might be empty
  596.      */
  597.     public function getPrefixedMarkers($prefix) {
  598.         $matches = array();
  599.         preg_match_all(
  600.             '/(#)('.strtoupper($prefix).'_[^#]+)/',
  601.             $this->markerNames, $matches
  602.         );
  603.  
  604.         $result = array_unique($matches[2]);
  605.  
  606.         return $result;
  607.     }
