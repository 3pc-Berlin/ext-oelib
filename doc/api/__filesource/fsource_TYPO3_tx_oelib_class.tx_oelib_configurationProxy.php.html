  25. /**
  26.  * Class 'tx_oelib_configurationProxy' for the 'oelib' extension.
  27.  *
  28.  * This singleton class provides access to an extension's global configuration
  29.  * and allows to fake global configuration values for testing purposes.
  30.  *
  31.  * @package        TYPO3
  32.  * @subpackage    tx_oelib
  33.  *
  34.  * @author        Saskia Metzler <saskia@merlin.owl.de>
  35.  */
  36. class tx_oelib_configurationProxy {
  37.     /** the singleton configuration proxy objects */
  38.     private static $instances = array();
  39.  
  40.     /**
  41.      * stored configuration data for each extension which currently uses the
  42.      * configuration proxy
  43.      */
  44.     private $configuration = array();
  45.  
  46.     /** key of the extension for which the EM configuration is stored */
  47.     private $extensionKey = '';
  48.  
  49.     /** whether the configuration is already loaded */
  50.     private $isConfigurationLoaded = false;
  51.  
  52.     /**
  53.      * Don't call this constructor; use getInstance instead.
  54.      *
  55.      * @param    string        extension key without the 'tx' prefix, used to
  56.      *                          retrieve the EM configuration and as identifier for
  57.      *                          an extension's instance of this class, must not be
  58.      *                          empty
  59.      */
  60.     private function __construct($extensionKey) {
  61.         $this->extensionKey = $extensionKey;
  62.     }
  63.  
  64.     /**
  65.      * Retrieves the singleton configuration proxy instance for the extension
  66.      * named $extensionKey. This function usually should be called statically.
  67.      *
  68.      * @param    string        extension key without the 'tx' prefix, used to
  69.      *                          retrieve the EM configuration and as identifier for
  70.      *                          an extension's instance of this class, must not be
  71.      *                          empty
  72.      *
  73.      * @return    tx_oelib_configurationProxy        the singleton configuration
  74.      *                          proxy object
  75.      */
  76.     public static function getInstance($extensionKey) {
  77.         if ($extensionKey == '') {
  78.             throw new Exception('The extension key was not set.');
  79.         }
  80.  
  81.         if (!is_object(self::$instances[$extensionKey])) {
  82.             self::$instances[$extensionKey]
  83.                 = new tx_oelib_configurationProxy($extensionKey);
  84.         }
  85.  
  86.         return self::$instances[$extensionKey];
  87.     }
  88.  
  89.     /**
  90.      * Loads the EM configuration for the extionsion key passed via
  91.      * getInstance() if the configuration is not yet loaded.
  92.      */
  93.     private function loadConfigurationLazily() {
  94.         if (!$this->isConfigurationLoaded) {
  95.             $this->retrieveConfiguration();
  96.         }
  97.     }
  98.  
  99.     /**
  100.      * Retrieves the EM configuration for the extionsion key passed via
  101.      * getInstance().
  102.      *
  103.      * This function is accessible for testing purposes. As lazy implementation
  104.      * is used, this function might be useful to ensure static test conditions.
  105.      */
  106.     public function retrieveConfiguration() {
  107.         $this->configuration = unserialize(
  108.             $GLOBALS['TYPO3_CONF_VARS']['EXT']['extConf'][$this->extensionKey]
  109.         );
  110.         $this->isConfigurationLoaded = true;
  111.     }
  112.  
  113.     /**
  114.      * Checks whether a certain key exists in an extension's configuration.
  115.      *
  116.      * @param    string        key to check, must not be empty
  117.      *
  118.      * @return    boolean        whether $key occurs in the configuration array of
  119.      *                          the extension named $this->extensionKey
  120.      */
  121.     private function hasConfigurationValue($key) {
  122.         $this->loadConfigurationLazily();
  123.  
  124.         return isset($this->configuration[$key]);
  125.     }
  126.  
  127.     /**
  128.      * Returns a configuration value.
  129.      *
  130.      * @param    string        key of the value to get, must not be empty
  131.      *
  132.      * @return    mixed        configuration value, will be an empty string if the
  133.      *                          value does not exist
  134.      */
  135.     private function getConfigurationValue($key) {
  136.         $this->loadConfigurationLazily();
  137.  
  138.         if ($this->hasConfigurationValue($key)) {
  139.             $result = $this->configuration[$key];
  140.         } else {
  141.             $result = '';
  142.         }
  143.  
  144.         return $result;
  145.     }
  146.  
  147.     /**
  148.      * Sets a new configuration value.
  149.      *
  150.      * The configuration setters are intended to be used for testing purposes
  151.      * only.
  152.      *
  153.      * @param    string        key of the value to set, must not be empty
  154.      * @param    mixed        value to set
  155.      */
  156.     private function setConfigurationValue($key, $value) {
  157.         $this->loadConfigurationLazily();
  158.  
  159.         $this->configuration[$key] = $value;
  160.     }
  161.  
  162.     /**
  163.      * Returns a string configuration value.
  164.      *
  165.      * @param    string        key of the value to get, must not be empty
  166.      *
  167.      * @return    string        configuration value string, might be empty
  168.      */
  169.     public function getConfigurationValueString($key) {
  170.         return trim((string) $this->getConfigurationValue($key));
  171.     }
  172.  
  173.     /**
  174.      * Sets a string configuration value.
  175.      *
  176.      * @param    string        key of the value to set, must not be empty
  177.      * @param    string        value to set, may be empty
  178.      */
  179.     public function setConfigurationValueString($key, $value) {
  180.         $this->setConfigurationValue($key, trim($value));
  181.     }
  182.  
  183.     /**
  184.      * Returns a boolean configuration value.
  185.      *
  186.      * @param    string        key of the value to get, must not be empty
  187.      *
  188.      * @return    boolean        boolean configuration value
  189.      */
  190.     public function getConfigurationValueBoolean($key) {
  191.         return (boolean) $this->getConfigurationValue($key);
  192.     }
  193.  
  194.     /**
  195.      * Sets a boolean configuration value.
  196.      *
  197.      * @param    string        key of the value to set, must not be empty
  198.      * @param    boolean        value to set
  199.      */
  200.     public function setConfigurationValueBoolean($key, $value) {
  201.         $this->setConfigurationValue($key, intval($value));
  202.     }
  203.  
  204.     /**
  205.      * Returns an integer configuration value.
  206.      *
  207.      * @param    string        key of the value to get, must not be empty
  208.      *
  209.      * @return    integer        configuration value
  210.      */
  211.     public function getConfigurationValueInteger($key) {
  212.         return intval($this->getConfigurationValue($key));
  213.     }
  214.  
  215.     /**
  216.      * Sets an integer configuration value.
  217.      *
  218.      * @param    string        key of the value to set, must not be empty
  219.      * @param    integer        value to set
  220.      */
  221.     public function setConfigurationValueInteger($key, $value) {
  222.         $this->setConfigurationValue($key, $value);
  223.     }
  224.  
  225.     /**
  226.      * Returns an extension's complete configuration.
  227.      *
  228.      * @return    array        an extension's configuration, empty if the
  229.      *                          configuration was not retrieved before
  230.      */
  231.     public function getCompleteConfiguration() {
  232.         $this->loadConfigurationLazily();
  233.  
  234.         return $this->configuration;
  235.     }
