  57.     /**
  58.      * Explodes a comma-separated list of integer values and sorts them
  59.      * numerically.
  60.      *
  61.      * @param    string        comma-separated list of values, may be empty
  62.      *
  63.      * @return    array        the separate values, sorted numerically, may be
  64.      *                          empty
  65.      */
  66.     private function sortExplode($valueList) {
  67.         if ($valueList == '') {
  68.             return array();
  69.         }
  70.  
  71.         $numbers = t3lib_div::intExplode(',', $valueList);
  72.         sort($numbers, SORT_NUMERIC);
  73.  
  74.         return ($numbers);
  75.     }
