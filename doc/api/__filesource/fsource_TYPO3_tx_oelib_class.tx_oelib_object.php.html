  42.     /**

  45.      * @param    string        the key of the data item to get, must not be empty

  73.     /**

  87.     /**

  144.     protected function setAsInteger($key, $value) {

  150.     /**

  169.     /**

  173.      * @param    string        the key of the element to retrieve, must not be

  187.  
  188.     /**
