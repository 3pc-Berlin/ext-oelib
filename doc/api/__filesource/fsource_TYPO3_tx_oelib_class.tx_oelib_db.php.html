  37. class tx_oelib_db {
  38.     /**
  39.      * @var    t3lib_pageSelect        page object which we will use to call

  41.      */
  42.     private static $pageForEnableFields = null;

  47.     private static $enableFieldsCache = array();

  49.     /**
  50.      * Wrapper function for t3lib_pageSelect::enableFields() since it is no
  51.      * longer accessible statically.
  52.      *
  53.      * Returns a part of a WHERE clause which will filter out records with
  54.      * start/end times or deleted/hidden/fe_groups fields set to values that
  55.      * should de-select them according to the current time, preview settings or
  56.      * user login.
  57.      * Is using the $TCA arrays "ctrl" part where the key "enablefields"
  58.      * determines for each table which of these features applies to that table.
  59.      *
  60.      * @param    string        table name found in the $TCA array
  61.      * @param    integer        If $showHidden is set (0/1), any hidden-fields in
  62.      *                          records are ignored. NOTICE: If you call this function,
  63.      *                          consider what to do with the show_hidden parameter.
  64.      *                          Maybe it should be set? See tslib_cObj->enableFields
  65.      *                          where it's implemented correctly.
  66.      * @param    array        Array you can pass where keys can be "disabled",
  67.      *                          "starttime", "endtime", "fe_group" (keys from
  68.      *                          "enablefields" in TCA) and if set they will make sure
  69.      *                          that part of the clause is not added. Thus disables
  70.      *                          the specific part of the clause. For previewing etc.
  71.      * @param    boolean        If set, enableFields will be applied regardless of
  72.      *                          any versioning preview settings which might otherwise
  73.      *                          disable enableFields.
  74.      *
  75.      * @return    string        the WHERE clause starting like
  76.      *                          " AND ...=... AND ...=..."
  77.      */
  78.     public static function enableFields(
  79.         $table, $showHidden = -1, array $ignoreArray = array(),
  80.         $noVersionPreview = false
  81.     ) {
  82.         if (!in_array($showHidden, array(-1, 0, 1))) {
  83.             throw new Exception(
  84.                 '$showHidden may only be -1, 0 or 1, but actually is ' .
  85.                     $showHidden
  86.             );
  87.         }
  88.  
  89.         // maps $showHidden (-1..1) to (0..2) which ensures valid array keys
  90.         $showHiddenKey = $showHidden + 1;
  91.         $ignoresKey = serialize($ignoreArray);
  92.         $previewKey = intval($noVersionPreview);
  93.         if (!isset(self::$enableFieldsCache[$table][$showHiddenKey][$ignoresKey]
  94.             [$previewKey])
  95.         ) {
  96.             self::retrievePageForEnableFields();
  97.             self::$enableFieldsCache[$table][$showHiddenKey][$ignoresKey]
  98.                 [$previewKey]
  99.                 = self::$pageForEnableFields->enableFields(
  100.                     $table,
  101.                     $showHidden,
  102.                     $ignoreArray,
  103.                     $noVersionPreview
  104.                 );
  105.         }
  106.  
  107.         return self::$enableFieldsCache[$table][$showHiddenKey][$ignoresKey]
  108.             [$previewKey];
  109.     }

  111.     /**
  112.      * Makes sure that self::$pageForEnableFields is a page object.
  113.      */
  114.     private static function retrievePageForEnableFields() {
  115.         if (!is_object(self::$pageForEnableFields)) {
  116.             if (isset($GLOBALS['TSFE'])
  117.                 && is_object($GLOBALS['TSFE']->sys_page)
  118.             ) {
  119.                 self::$pageForEnableFields = $GLOBALS['TSFE']->sys_page;
  120.             } else {
  121.                 self::$pageForEnableFields
  122.                     = t3lib_div::makeInstance('t3lib_pageSelect');
  123.             }
  124.         }
  125.     }
  126.  
  127.     /**
  128.      * Recursively creates a comma-separated list of subpage UIDs from
  129.      * a list of pages. The result also includes the original pages.
  130.      * The maximum level of recursion can be limited:
  131.      * 0 = no recursion (the default value, will return $startPages),
  132.      * 1 = only direct child pages,
  133.      * ...,
  134.      * 250 = all descendants for all sane cases
  135.      *
  136.      * Note: The returned page list is _not_ sorted.
  137.      *
  138.      * @param    string        comma-separated list of page UIDs to start from,
  139.      *                          must only contain numbers and commas, may be empty
  140.      * @param    integer        maximum depth of recursion, must be >= 0
  141.      *
  142.      * @return    string        comma-separated list of subpage UIDs including the
  143.      *                          UIDs provided in $startPages, will be empty if
  144.      *                          $startPages is empty
  145.      */
  146.     public static function createRecursivePageList(
  147.         $startPages, $recursionDepth = 0
  148.     ) {
  149.         if ($recursionDepth < 0) {
  150.             throw new Exception('$recursionDepth must be >= 0.');
  151.         }
  152.         if ($recursionDepth == 0) {
  153.             return $startPages;
  154.         }
  155.         if ($startPages == '') {
  156.             return '';
  157.         }
  158.  
  159.         $dbResult = $GLOBALS['TYPO3_DB']->exec_SELECTquery(
  160.             'uid',
  161.             'pages',
  162.             'pid IN (' . $startPages . ')' . tx_oelib_db::enableFields('pages')
  163.         );
  164.         if (!$dbResult) {
  165.             throw new Exception(DATABASE_QUERY_ERROR);
  166.         }
  167.  
  168.         $subPages = array();
  169.         while ($row = $GLOBALS['TYPO3_DB']->sql_fetch_assoc($dbResult)) {
  170.             $subPages[] = $row['uid'];
  171.         }
  172.         $GLOBALS['TYPO3_DB']->sql_free_result($dbResult);
  173.  
  174.         if (!empty($subPages)) {
  175.             $result = $startPages . ',' . self::createRecursivePageList(
  176.                 implode(',', $subPages), $recursionDepth - 1
  177.             );
  178.         } else {
  179.             $result = $startPages;
  180.         }
  181.         return $result;
  182.     }
  183. }
