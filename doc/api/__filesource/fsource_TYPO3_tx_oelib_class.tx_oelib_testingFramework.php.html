  536.     /**
  537.      * Changes an existing dummy record and stores the new data for this
  538.      * record. Only fields that get new values in $recordData will be changed,
  539.      * everything else will stay untouched.
  540.      *
  541.      * The array with the new recordData must contain at least one entry, but
  542.      * must not contain a new UID for the record. If you need to change the UID,
  543.      * you have to create a new record!
  544.      *
  545.      * @param    string        the name of the table, must not be empty
  546.      * @param    integer        the UID of the record to change, must not be empty
  547.      * @param    array        associative array containing key => value pairs for
  548.      *                          those fields of the record that need to be changed,
  549.      *                          must not be empty
  550.      */
  551.     public function changeRecord($table, $uid, $recordData) {
  552.         $dummyColumnName = $this->getDummyColumnName($table);
  553.  
  554.         if (!$this->isTableNameAllowed($table)) {
  555.             throw new Exception(
  556.                 'The table "' . $table . '" is not on the lists with allowed tables.'
  557.             );
  558.         }
  559.         if ($uid == 0) {
  560.             throw new Exception('The parameter $uid must not be zero.');
  561.         }
  562.         if (empty($recordData)) {
  563.             throw new Exception(
  564.                 'The array with the new record data must not be empty.'
  565.             );
  566.         }
  567.         if (isset($recordData['uid'])) {
  568.             throw new Exception(
  569.                 'The parameter $recordData must not contain changes to the UID' .
  570.                 ' of a record.'
  571.             );
  572.         }
  573.         if (isset($recordData[$dummyColumnName])) {
  574.             throw new Exception(
  575.                 'The parameter $recordData must not contain changes to the ' .
  576.                     'field "' . $dummyColumnName . '". It is impossible to ' .
  577.                     'convert a dummy record into a regular record.'
  578.             );
  579.         }
  580.         if (!$this->countRecords($table, 'uid='.$uid)) {
  581.             throw new Exception(
  582.                 'There is no record with UID ' . $uid . ' on table "' . $table . '".'
  583.             );
  584.         }
  585.  
  586.         $dbResult = $GLOBALS['TYPO3_DB']->exec_UPDATEquery(
  587.             $table,
  588.             'uid=' . $uid . ' AND ' . $dummyColumnName . '=1',
  589.             $recordData
  590.         );
  591.  
  592.         if (!$dbResult) {
  593.             throw new Exception(DATABASE_QUERY_ERROR);
  594.         }
  595.     }
  596.  
  597.     /**
  598.      * Deletes a dummy record from the database.
  599.      *
  600.      * Important: Only dummy records from non-system tables can be deleted with
  601.      * this method. Should there for any reason exist a real record with that
  602.      * UID, it won't be deleted.
  603.      *
  604.      * @param    string        name of the table from which the record should be
  605.      *                          deleted, must not be empty
  606.      * @param    integer        UID of the record to delete, must be > 0
  607.      */
  608.     public function deleteRecord($table, $uid) {
  609.         if (!$this->isNoneSystemTableNameAllowed($table)) {
  610.             throw new Exception('The table name "' . $table . '" is not allowed.');
  611.         }
  612.  
  613.         $dbResult = $GLOBALS['TYPO3_DB']->exec_DELETEquery(
  614.             $table,
  615.             'uid=' . $uid . ' AND ' . $this->getDummyColumnName($table) . '=1'
  616.         );
  617.  
  618.         if (!$dbResult) {
  619.             throw new Exception(DATABASE_QUERY_ERROR);
  620.         }
  621.     }
  622.  
  623.     /**
  624.      * Creates a relation between two records on different tables (so called
  625.      * m:n relation).
  626.      *
  627.      * @param    string        name of the m:n table to which the record should be
  628.      *                          added, must not be empty
  629.      * @param    integer        UID of the local table, must be > 0
  630.      * @param    integer        UID of the foreign table, must be > 0
  631.      * @param    integer        sorting value of the relation, the default value is
  632.      *                          0, which enables automatic sorting, a value >= 0
  633.      *                          overwrites the automatic sorting
  634.      */
  635.     public function createRelation($table, $uidLocal, $uidForeign, $sorting = 0) {
  636.         if (!$this->isNoneSystemTableNameAllowed($table)) {
  637.             throw new Exception('The table name "' . $table . '" is not allowed.');
  638.         }
  639.  
  640.         // Checks that the two given UIDs are valid.
  641.         if (intval($uidLocal) <= 0) {
  642.             throw new Exception(
  643.                 '$uidLocal must be an integer > 0, but actually is "' .
  644.                     $uidLocal . '"'
  645.             );
  646.         }
  647.         if  (intval($uidForeign) <= 0) {
  648.             throw new Exception(
  649.                 '$uidForeign must be an integer > 0, but actually is "' .
  650.                     $uidForeign . '"'
  651.             );
  652.         }
  653.  
  654.         $this->markTableAsDirty($table);
  655.  
  656.         $recordData = array(
  657.             'uid_local' => $uidLocal,
  658.             'uid_foreign' => $uidForeign,
  659.             'sorting' => (($sorting > 0) ?
  660.                 $sorting : $this->getRelationSorting($table, $uidLocal)),
  661.             $this->getDummyColumnName($table) => 1
  662.         );
  663.  
  664.         // Stores the record in the database.
  665.         $dbResult = $GLOBALS['TYPO3_DB']->exec_INSERTquery(
  666.             $table,
  667.             $recordData
  668.         );
  669.  
  670.         if (!$dbResult) {
  671.             throw new Exception(DATABASE_QUERY_ERROR);
  672.         }
  673.     }
  674.  
  675.     /**
  676.      * Deletes a dummy relation from an m:n table in the database.
  677.      *
  678.      * Important: Only dummy records can be deleted with this method. Should there
  679.      * for any reason exist a real record with that combination of local and
  680.      * foreign UID, it won't be deleted!
  681.      *
  682.      * @param    string        name of the table from which the record should be
  683.      *                          deleted, must not be empty
  684.      * @param    integer        UID on the local table, must be > 0
  685.      * @param    integer        UID on the foreign table, must be > 0
  686.      */
  687.     public function removeRelation($table, $uidLocal, $uidForeign) {
  688.         if (!$this->isNoneSystemTableNameAllowed($table)) {
  689.             throw new Exception('The table name "' . $table . '" is not allowed.');
  690.         }
  691.  
  692.         $dbResult = $GLOBALS['TYPO3_DB']->exec_DELETEquery(
  693.             $table,
  694.             'uid_local=' . $uidLocal . ' AND uid_foreign=' . $uidForeign .
  695.                 ' AND ' . $this->getDummyColumnName($table) . '=1'
  696.         );
  697.  
  698.         if (!$dbResult) {
  699.             throw new Exception(DATABASE_QUERY_ERROR);
  700.         }
  701.     }
  702.  
  703.     /**
  704.      * Deletes all dummy records that have been added through this framework.
  705.      * For this, all records with the "is_dummy_record" flag set to 1 will be
  706.      * deleted from all tables that have been used within this instance of the
  707.      * testing framework.
  708.      *
  709.      * If you set $performDeepCleanUp to true, it will go through ALL tables to
  710.      * which the current instance of the testing framework has access. Please
  711.      * consider well, whether you want to do this as it's a huge performance
  712.      * issue.
  713.      *
  714.      * @param    boolean        whether a deep clean up should be performed, may
  715.      *                          be empty
  716.      */
  717.     public function cleanUp($performDeepCleanUp = false) {

  719.         $this->cleanUpTableSet(true, $performDeepCleanUp);

  722.         $this->discardFakeFrontEnd();
  723.     }

  725.     /**
  726.      * Deletes a set of records that have been added through this framework for
  727.      * a set of tables (either the test tables or the allowed system tables).
  728.      * For this, all records with the "is_dummy_record" flag set to 1 will be
  729.      * deleted from all tables that have been used within this instance of the
  730.      * testing framework.
  731.      *
  732.      * If you set $performDeepCleanUp to true, it will go through ALL tables to
  733.      * which the current instance of the testing framework has access. Please
  734.      * consider well, whether you want to do this as it's a huge performance
  735.      * issue.
  736.      *
  737.      * @param    boolean        whether to clean up the system tables (true) or
  738.      *                          the non-system test tables (false)
  739.      * @param    boolean        whether a deep clean up should be performed, may
  740.      *                          be empty
  741.      */
  742.     private function cleanUpTableSet($useSystemTables, $performDeepCleanUp) {
  743.         if ($useSystemTables) {
  744.             $tablesToCleanUp = ($performDeepCleanUp)
  745.                 ? $this->allowedSystemTables
  746.                 : $this->dirtySystemTables;
  747.         } else {
  748.             $tablesToCleanUp = ($performDeepCleanUp)
  749.                 ? $this->ownAllowedTables
  750.                 : $this->dirtyTables;
  751.         }
  752.  
  753.         foreach ($tablesToCleanUp as $currentTable) {
  754.             $dummyColumnName = $this->getDummyColumnName($currentTable);
  755.  
  756.             // Runs a delete query for each allowed table. A "one-query-deletes-them-all"
  757.             // approach was tested but we didn't find a working solution for that.
  758.             $dbResult = $GLOBALS['TYPO3_DB']->exec_DELETEquery(
  759.                 $currentTable,
  760.                 $dummyColumnName.'=1'
  761.             );
  762.  
  763.             // Resets the auto increment setting of the current table.
  764.             $this->resetAutoIncrementLazily($currentTable);
  765.  
  766.             if (!$dbResult) {
  767.                 throw new Exception(DATABASE_QUERY_ERROR);
  768.             }
  769.         }
  770.  
  771.         // Resets the list of dirty tables.
  772.         $this->dirtyTables = array();
  773.     }

  797.  
  798.     /**
  799.      * Creates an empty dummy file with a unique file name in the calling
  800.      * extension's upload directory.
  801.      *
  802.      * @param    string        path of the dummy file to create, relative to the
  803.      *                          calling extension's upload directory, must not be
  804.      *                          empty
  805.      *
  806.      * @return    string        the absolute path of the created dummy file, will
  807.      *                          not be empty
  808.      */
  809.     public function createDummyFile($fileName = 'test.txt') {
  810.         $uniqueFileName = $this->getUniqueFileOrFolderPath($fileName);
  811.  
  812.         if (!@t3lib_div::writeFile($uniqueFileName, '')) {
  813.             throw new Exception($uniqueFileName . ' could not be created.');
  814.         }
  815.  
  816.         $relativeFileName = $this->getPathRelativeToUploadDirectory(
  817.             $uniqueFileName
  818.         );
  819.
