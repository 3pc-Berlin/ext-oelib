  31. /**
  32.  * Class 'tx_oelib_dataMapper' for the 'oelib' extension.
  33.  *
  34.  * This class represents a mapper that maps database record to model instances.
  35.  *
  36.  * @package        TYPO3
  37.  * @subpackage    tx_oelib
  38.  *
  39.  * @author        Oliver Klee <typo3-coding@oliverklee.de>
  40.  */
  41. abstract class tx_oelib_dataMapper {
  42.     /**
  43.      * @var    string        the name of the database table for this mapper,
  44.      *                      must not be empty in subclasses
  45.      */
  46.     protected $tableName = '';
  47.  
  48.     /**
  49.      * @var    string        a comma-separated list of DB column names to retrieve
  50.      *                      or "*" for all columns, must not be empty
  51.      */
  52.     protected $columns = '*';
  53.  
  54.     /**
  55.      * @var    tx_oelib_identityMap    a map that holds the models that already
  56.      *                                  have been retrieved
  57.      */
  58.     protected $map = null;
  59.  
  60.     /**
  61.      * The constructor.
  62.      */
  63.     public function __construct() {
  64.         if ($this->tableName == '') {
  65.             throw new Exception(
  66.                 get_class($this) . '::tableName must not be empty.'
  67.             );
  68.         }
  69.         if ($this->columns == '') {
  70.             throw new Exception(
  71.                 get_class($this) . '::columns must not be empty.'
  72.             );
  73.         }
  74.  
  75.         $this->map = t3lib_div::makeInstance('tx_oelib_identityMap');
  76.     }
  77.  
  78.     /**
  79.      * Frees as much memory that has been used by this object as possible.
  80.      */
  81.     public function __destruct() {
  82.         if ($this->map) {
  83.             $this->map->__destruct();
  84.             unset($this->map);
  85.         }
  86.     }
  87.  
  88.     /**
  89.      * Retrieves a record from the DB by UID and creates a model from it. If
  90.      * the model already is cached in memory, the cached instance is returned.
  91.      *
  92.      * @throws    tx_oelib_notFoundException    if there is no record in the DB
  93.      *                                          with that particular UID
  94.      *
  95.      * @param    integer                        the UID of the record to retrieve,
  96.      *                                          must be > 0
  97.      *
  98.      * @return    tx_oelib_model                the stored model with the UID $uid
  99.      */
  100.     public function find($uid) {
  101.         try {
  102.             $result = $this->map->get($uid);
  103.             return $result;
  104.         } catch (tx_oelib_notFoundException $exception) {
  105.             $result = $this->load($uid);
  106.             return $result;
  107.         }
  108.     }
  109.  
  110.     /**
  111.      * Retrieves a record from the DB without consulting this mapper's map,
  112.      * and then stores it in the map.
  113.      *
  114.      * @throws    tx_oelib_notFoundException    if there is no record in the DB
  115.      *                                          with that particular UID
  116.      *
  117.      * @param    integer                        the UID of the record to retrieve,
  118.      *                                          must be > 0
  119.      *
  120.      * @return    tx_oelib_model                the stored model with the UID $uid
  121.      */
  122.     protected function load($uid) {
  123.         $queryResult = $GLOBALS['TYPO3_DB']->exec_SELECTquery(
  124.             $this->columns,
  125.             $this->tableName,
  126.             'uid = ' . $uid . tx_oelib_db::enableFields($this->tableName)
  127.         );
  128.         if (!$queryResult) {
  129.             throw new Exception(DATABASE_QUERY_ERROR);
  130.         }
  131.  
  132.         $data = $GLOBALS['TYPO3_DB']->sql_fetch_assoc($queryResult);
  133.         $GLOBALS['TYPO3_DB']->sql_free_result($queryResult);
  134.  
  135.         if (!$data) {
  136.             throw new tx_oelib_notFoundException(
  137.                 'The record with the UID ' . $uid . ' could not be retrieved ' .
  138.                     'from the table ' . $this->tableName . '.'
  139.             );
  140.         }
  141.  
  142.         $model = $this->createAndFillModel($data);
  143.  
  144.         $this->map->add($model);
  145.  
  146.         return $model;
  147.     }
  148.  
  149.     /**
  150.      * Creates a model of the correct type for this mapper and fills it with
  151.      * the data provided as $data.
  152.      *
  153.      * @param    array                the data with which the model should be
  154.      *                                  filled, may be empty
  155.      *
  156.      * @return    tx_oelib_model        the filled model
  157.      */
  158.     protected abstract function createAndFillModel(array $data);
  159. }
