  93.     /**
  94.      * Retrieves the localized string for the local language key $key.
  95.      *
  96.      * This function checks whether the FE or BE localization functions are
  97.      * available and then uses the appropriate method.
  98.      *
  99.      * In $this->conf['salutation'], a suffix to the key may be set (which may
  100.      * be either 'formal' or 'informal'). If a corresponding key exists, the
  101.      * formal/informal localized string is used instead.
  102.      * If the formal/informal key doesn't exist, this function just uses the
  103.      * regular string.
  104.      *
  105.      * Example: key = 'greeting', suffix = 'informal'. If the key
  106.      * 'greeting_informal' exists, that string is used.
  107.      * If it doesn't exist, this functions tries to use the string with the key
  108.      * 'greeting'.
  109.      *
  110.      * @param    string        the local language key for which to
  111.      *                          return the value, must not be empty
  112.      * @param    boolean        whether the output should be passed through
  113.      *                          htmlspecialchars()
  114.      *
  115.      * @return    string        the requested local language key, might be empty
  116.      */
  117.     public function translate(
  118.         $key, $useHtmlSpecialChars = false
  119.     ) {
  120.         if ($key == '') {
  121.             throw new Exception('$key must not be empty.');
  122.         }
  123.  
  124.         if (is_object($GLOBALS['TSFE']) && is_array($this->LOCAL_LANG)) {
  125.             $result = $this->translateInFrontEnd($key);
  126.         } elseif (is_object($GLOBALS['LANG'])) {
  127.             $result = $this->translateInBackEnd($key);
  128.         } else {
  129.             $result = $key;
  130.         }
  131.  
  132.         if ($useHtmlSpecialChars) {
  133.             $result = htmlspecialchars($result);
  134.         }
  135.  
  136.         return $result;
  137.     }
  138.  
  139.     /**
  140.      * Retrieves the localized string for the local language key $key, using the
  141.      * BE localization methods.
  142.      *
  143.      * @param    string        the local language key for which to
  144.      *                          return the value, must not be empty

  146.      * @return    string        the requested local language key, might be empty

  148.     private function translateInBackEnd($key) {

  150.     }

  152.     /**
  153.      * Retrieves the localized string for the local language key $key, using the
  154.      * FE localization methods.
  155.      *
  156.      * In $this->conf['salutation'], a suffix to the key may be set (which may
  157.      * be either 'formal' or 'informal'). If a corresponding key exists, the
  158.      * formal/informal localized string is used instead.
  159.      * If the formal/informal key doesn't exist, this function just uses the
  160.      * regular string.
  161.      *
  162.      * Example: key = 'greeting', suffix = 'informal'. If the key
  163.      * 'greeting_informal' exists, that string is used.
  164.      * If it doesn't exist, this functions tries to use the string with the key
  165.      * 'greeting'.
  166.      *
  167.      * @param    string        the local language key for which to
  168.      *                          return the value, must not be empty
  169.      *
  170.      * @return    string        the requested local language key, might be empty
  171.      */
  172.     private function translateInFrontEnd($key) {
  173.         $hasFoundATranslation = false;
  174.  
  175.         $availableLanguages = $this->getAvailableLanguages();
  176.         $suffixesToTry = $this->getSuffixesToTry();
  177.  
  178.         foreach ($availableLanguages as $language) {
  179.             foreach ($suffixesToTry as $suffix) {
  180.                 $completeKey = $key.$suffix;
  181.                 if (isset($this->LOCAL_LANG[$language][$completeKey])) {
  182.                     $result = parent::pi_getLL($completeKey);
  183.                     $hasFoundATranslation = true;
  184.                     break 2;
  185.                 }
  186.             }
  187.         }
  188.  
  189.         // If still nothing has been found, just return the key.
  190.         if (!$hasFoundATranslation) {
  191.             $result = $key;
  192.         }
  193.  
  194.         return $result;
  195.     }
  196.  
  197.     /**
  198.      * Compiles a list of language keys for which localizations have been loaded.
  199.      *
  200.      * @return    array        a list of language keys (may be empty)
  201.      */
  202.     private function getAvailableLanguages() {
  203.         if ($this->availableLanguages === null) {
  204.             $this->availableLanguages = array();
  205.  
  206.             if (!empty($this->LLkey)) {
  207.                 $this->availableLanguages[] = $this->LLkey;
  208.             }
  209.             if (!empty($this->altLLkey)) {
  210.                 $this->availableLanguages[] = $this->altLLkey;
  211.             }
  212.             // The key for English is "default", not "en".
  213.             $this->availableLanguages = preg_replace(
  214.                 '/en/', 'default', $this->availableLanguages
  215.             );
  216.             // Remove duplicates in case the default language is the same as the
  217.             // fall-back language.
  218.             $this->availableLanguages = array_unique($this->availableLanguages);
  219.  
  220.             // Now check that we only keep languages for which we have
  221.             // translations.
  222.             foreach ($this->availableLanguages as $index => $code) {
  223.                 if (!isset($this->LOCAL_LANG[$code])) {
  224.                     unset($this->availableLanguages[$index]);
  225.                 }
  226.             }
  227.         }
  228.  
  229.         return $this->availableLanguages;
  230.     }
  231.  
  232.     /**
  233.      * Gets an ordered list of language label suffixes that should be tried to
  234.      * get localizations in the preferred order of formality.
  235.      *
  236.      * @return    array        ordered list of suffixes from "", "_formal" and "_informal", will not be empty
  237.      */
  238.     private function getSuffixesToTry() {
  239.         if ($this->suffixesToTry === null) {
  240.             $this->suffixesToTry = array();
  241.  
  242.             if (isset($this->conf['salutation'])
  243.                 && ($this->conf['salutation'] == 'informal')) {
  244.                 $this->suffixesToTry[] = '_informal';
  245.             }
  246.             $this->suffixesToTry[] = '_formal';
  247.             $this->suffixesToTry[] = '';
  248.         }
  249.  
  250.         return $this->suffixesToTry;
  251.     }
  252. }
